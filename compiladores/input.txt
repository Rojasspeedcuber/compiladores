int main() {
    int x = 5;
    int y = 10;

    // Expressões Aritméticas
    int soma = x + y;
    int subtracao = x - y;
    int multiplicacao = x * y;
    int divisao = x / y;
    int resto = x % y;

    // Expressões Relacionais
    bool maior = x > y;
    bool menor_ou_igual = x <= y;
    bool igualdade = x == y;
    bool diferenca = x != y;

    // Expressões Lógicas
    bool and_logico = (x > 0) && (y > 0);
    bool or_logico = (x > 0) || (y > 0);
    bool not_logico = !(x > 0);

    // Expressões Unárias
    int negativo = -x;
    int positivo = +x;

    // Expressões Pós-fixas
    int array[5] = {1, 2, 3, 4, 5};
    int elemento = array[2];
    int resultado_funcao = funcao_teste(3);

    return 0;
}

int funcao_teste(int parametro) {
    return parametro * 2;
}
